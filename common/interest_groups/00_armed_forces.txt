ig_armed_forces = {
	color = hsv{ 0.09 0.29 0.39 }
	texture = "gfx/interface/icons/ig_icons/armed_forces.dds"
	
	ideologies = {
		ideology_jingoist
		ideology_loyalist
		ideology_patriotic
	}
	
	enable = {
		always = yes
	}
	
	traits = {
		ig_trait_patriotic_fervor
		ig_trait_veteran_consultation
		ig_trait_materiel_waste
	}	
	
	on_enable = {}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
		OR = {
			is_pop_type = soldiers
			is_pop_type = officers
			is_pop_type = aristocrats
			pop_is_discriminated = no
		}		
	}
	
	pop_weight = {
		value = 0
		
		add = {
			desc = "POP_NOT_DISCRIMINATED"
			if = {
				limit = { 
					pop_is_discriminated = no
				}
				value = 20
			}
		}

		add = {
			desc = "IG_HAS_RIVAL"
			if = {
				limit = {
					owner = {
						any_rival_country = {
							count = 0
						}
					}
				}
				value = 0
			}
			else_if = {
				limit = {
					owner = {
						any_rival_country = {
							count = 1
						}
					}
				}
				value = 5
			}
			else_if ={
				limit = {
					owner = {
						any_rival_country = {
							count = 2
						}
					}
				}
				value = 10
			}
			else_if ={
				limit = {
					owner = {
						any_rival_country = {
							count = 3
						}
					}
				}
				value = 15
			}
			else_if ={
				limit = {
					owner = {
						any_rival_country = {
							count >= 4
						}
					}
				}
				value = 20
			}
		}

		add = {
			desc = "IG_EMPLOYED_MIC"
			if = {
				limit = {
					OR = {
						pop_employment_building = building_war_machine_industry
						pop_employment_building = building_arms_industry
						pop_employment_building = building_munition_plants
						AND = {
							pop_employment_building = building_shipyards
							workplace = {
								OR = {
									has_active_production_method = pm_military_shipbuilding_wooden
									has_active_production_method = pm_military_shipbuilding_wooden_2
									has_active_production_method = pm_military_shipbuilding_steam
									has_active_production_method = pm_military_shipbuilding_steam_2
								}
							}
						}
					}
				}
				value = 50
			}
		}

		add = {
			desc = "MIC_OWNER"
			if = {
				limit ={	
					is_pop_type = capitalists 
					OR = { 
						pop_employment_building = building_arms_industry
						pop_employment_building = building_munition_plants
						pop_employment_building = building_war_machine_industry
						AND = {
							pop_employment_building = building_shipyards
							workplace = {
								OR = {
									has_active_production_method = pm_military_shipbuilding_wooden
									has_active_production_method = pm_military_shipbuilding_wooden_2
									has_active_production_method = pm_military_shipbuilding_steam
									has_active_production_method = pm_military_shipbuilding_steam_2
								}
							}
						}
					}
				}
				value = 125
			}
		}
		
		add = {
			desc = "POP_SOLDIERS"
			if = {
				limit = { 
					is_pop_type = soldiers
				}
				value = 150
			}
		}
		
		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = { 
					is_pop_type = aristocrats
				}
				value = 100
			}
		}		
	
		add = {
			desc = "POP_OFFICERS"
			if = {
				limit = { 
					is_pop_type = officers
				}
				value = 200
			}
		}

		add = { #Policemens
			desc = "POP_CLERKS"
			if = {
				limit = { 
					is_pop_type = clerks
				}
				value = 25
			}
		}

		add = { #Policemens
			desc = "POP_BUREAUCRATS"
			if = {
				limit = { 
					is_pop_type = bureaucrats
				}
				value = 10
			}
		}

		add = {
			desc = "JUNKERS"
			if = {
				limit = { 
					is_pop_type = aristocrats
					owner = {
						ig:ig_landowners = {
							has_ideology = ideology_junker_paternalistic
						}
					}
				}
				value = 100
			}
		}

		add = {
			desc = "ideology_caudillismo"
			if = {
				limit = { 
					OR = {
						is_pop_type = officers
						is_pop_type = soldiers
						is_pop_type = aristocrats
					}
					owner = {
						ig:ig_armed_forces = {
							has_ideology = ideology_caudillismo
						}
					}
				}
				value = 250
			}
		}

		add = {
			desc = "BAKUFU"
			if = {
				limit = {
					is_pop_type = aristocrats
					owner = {
						ig:ig_armed_forces = {
							has_ideology = ideology_bakufu
						}
					}
				}
				value = 150
			}
		}

		multiply = { #Pops are less attracted to radical ideologies if they're not poor
			desc = "RADICAL_LEADER"
			value = 1.0
			if = {
				limit = {
					owner = {					
						ig:ig_armed_forces = {
							leader = {
								has_ideology = ideology_fascist
							}
						}
						NAND = {
							has_law = law_type:law_autocracy
							ig:ig_armed_forces = {
								is_in_government = yes
							}
						}						
					}
					NOT = {
						standard_of_living <= 14
					}
				}
				value = 0.75
			}
		}

		multiply = { #French army is Napoleon's nostalgic
			desc = "BONAPARTISTS"
			value = 1.0
			if = {
				limit = {
					OR = {
						is_pop_type = officers
						is_pop_type = soldiers
					}
					owner = {
						ig:ig_armed_forces = {
							has_ideology = ideology_bonapartism
						}
					}
				}
				value = 1.25
			}
		}

		multiply = {
			desc = "law_national_guard"
			value = 1.0
			if = {
				limit = {
					owner = {
						has_law = law_type:law_national_guard
					}
				}
				value = 1.1
			}
		}

		multiply = {
			desc = "LEFTISTARMY"
			value = 1.0

			if = {
				limit = {
					owner = {
						any_interest_group = { 
							is_interest_group_type = ig_armed_forces 
							leader = {
								OR = {
									has_ideology = ideology_communist
									has_ideology = ideology_vanguardist
									has_ideology = ideology_anarchist
									has_ideology = ideology_republican_leader
								}
							}
						}
					}
					is_pop_type = aristocrats
				}
				value = 0.1
			}
		}	

		multiply = {
			desc = "WOMENVOTE"
			value = 1.0

			if = {
				limit = {
					owner = {
						has_law = law_type:law_womens_suffrage
					}
				}
				value = 0.75
			}
		}
					
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}

		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}

		multiply = {
			desc = "IG_IS_MARGINALIZED"
			value = 1.0
			
			scope:interest_group = {
				if = {
					limit = {
						is_marginal = yes
					}
					value = 0.5
				}
			}
		}
	}
	
	monarch_weight = {
		value = 0.25
	}
	
	commander_weight = {
		value = 5.0
	}
	
	noble_chance = {
		value = 0.5
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0
	}	
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.75
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
