je_july_monarchy = {
	
	icon = "gfx/interface/icons/event_icons/event_protest.dds"
	
	immediate = {}
	
	timeout = 5111
	
	on_timeout = { trigger_event = { id = french_politics.1 popup = yes } }
	
	weight = 1000
	
	progressbar = no
	
	should_be_pinned_by_default = yes
	
	on_complete = {
		custom_tooltip = je_july_monarchy_goal.1
		hidden_effect = {
			if = {
				limit = {
					has_law = law_type:law_presidential_republic
				}
				trigger_event = { id = french_politics.2 popup = yes }
			}	
			else_if = {
				limit = {
					year >= 1847 
					AND = { 
						ig:ig_intelligentsia = { 
							#is_in_government = yes
							is_marginalized = no
						}	
					}
					NOT = { year > 1849 }
				}
				trigger_event = { id = french_politics.3 popup = yes }
			}
			else_if = {
				limit = {
					AND = {
						is_player = no
						year >= 1848
					}
				}
				trigger_event = { id = french_politics.3 popup = yes }
			}
			else_if = {
				limit = {
					any_civil_war = {
						civil_war_progress > 0.8
						is_civil_war_type = revolution
					}
					ig:ig_intelligentsia = { 
						is_insurrectionary = yes 
					}
				}
				trigger_event = { id = french_politics.3 popup = yes }
			}
		}
	}	
	
	complete = {
		exists = c:FRA	
		c:FRA = ROOT
		#is_at_war = no
		#NOT = { any_interest_group = { is_insurrectionary = yes } }	
		OR = { 
			has_law = law_type:law_presidential_republic
			AND = {
				year >= 1847
				ig:ig_intelligentsia = { 
					OR = {
						is_in_government = yes
						is_marginalized = no
					}
				}
				NOT = { year > 1849 }
			}
			hidden_trigger = {
				AND = {
	                is_player = no
					year >= 1848
	            }
            }	
		}
	}
	
	invalid = {
		OR = {
			any_scope_character = { 
				is_ruler = yes
				NOT = { has_variable = is_orleanist } 
			}
			AND = {
				NOT = { has_law = law_type:law_monarchy }
				NOT = { has_law = law_type:law_presidential_republic }
			}
		}
	}
}

je_future_of_the_republic = {
	
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	
	immediate = {
		set_variable = {
			name = future_of_the_republic_timer
			value = 0
		}
		random_scope_character = {
			limit = {
				has_variable = is_louis_napoleon
			}
			save_scope_as = louis_napoleon_leader
		}
	}
	
	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = future_of_the_republic_timer
				add = 1
			}
		}
	}
	
	timeout = 2191
	
	on_timeout = { trigger_event = { id = french_politics.7 popup = yes } }
	
	weight = 1000
	
	progressbar = no
	
	should_be_pinned_by_default = yes
	
	on_complete = {
		trigger_event = { id = french_politics.5 popup = yes }
	}	
	
	complete = {
		exists = c:FRA	
		c:FRA = ROOT
		is_at_war = no
		NOT = { any_interest_group = { is_insurrectionary = yes } }	
		OR = { 
			custom_tooltip = {
				text = je_future_of_the_republic_goal.1
				root.var:future_of_the_republic_timer >= 24				
				AND = {
					OR = {
						ig:ig_armed_forces = { 
							is_in_government = yes
							ig_clout >= 0.15
						}
						ig:ig_petty_bourgeoisie = { 
							is_in_government = yes
							ig_clout >= 0.15
						}
						ig:ig_rural_folk = { 
							is_in_government = yes
							ig_clout >= 0.15
						}
						AND = {
							ig:ig_armed_forces = { 
								is_in_government = yes
								ig_clout >= 0.10
							}
							ig:ig_petty_bourgeoisie = { 
								is_in_government = yes
								ig_clout >= 0.10
							}
						}
					}
					ruler = {
						has_variable = is_louis_napoleon
					}
					NOT = { root.var:future_of_the_republic_timer >= 49	}
				}
			}
			custom_tooltip = {
				text = je_future_of_the_republic_goal.2 
				AND ={ 
					ig:ig_intelligentsia = { 
						is_marginal = yes
					}
					ig:ig_trade_unions = {
						is_marginal = yes
					}
					ruler = {
						has_variable = is_louis_napoleon
					}
				}
			}	
			hidden_trigger = {
				AND = {
	                is_player = no
					root.var:future_of_the_republic_timer >= 24	
					NOT = { root.var:future_of_the_republic_timer >= 49	}
					ruler = {
						has_variable = is_louis_napoleon
					}
	            }
            }	
		}
	}
	
	invalid = {
		OR = {
			NOT = { has_law = law_type:law_presidential_republic }
			NOT = { any_scope_character = { has_variable = is_louis_napoleon } }
		}
	}
}
