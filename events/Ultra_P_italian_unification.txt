italian_independence_war.2 = { #Italians Wins
    type = country_event
    placement = ROOT

    title = italian_independence_war.2.t
    desc = italian_independence_war.2.d
    flavor = italian_independence_war.2.f

    event_image = {
        video = "gfx/event_pictures/europenorthamerica_political_extremism.bk2"
    }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds"
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    trigger = {
        AND = {
            exists = c:VEN
            OR = {
            	exists = c:SAR
            	exists = c:ITA
            }
        }
        THIS = c:VEN
        c:VEN = {
            is_at_war = no
            is_subject = no
        }
    }
    option = { #Ask for being annexed by Sardinia-Piedmont (Or Italy if it already exists)
        name = italian_independence_war.2.a
        if = {
        	limit = {
        		exists = c:ITA
        	}
        	c:ITA = {
	        	trigger_event = { 
					id = italian_independence_war.3
					popup = yes
				}
			}
        }
        else_if = {
        	limit = {
        		exists = c:SAR
        	}
        	c:SAR = {
	        	trigger_event = { 
					id = italian_independence_war.3
					popup = yes
				}
			}
        }
        ai_chance = {
            base = 100
        }
    } 
}

italian_independence_war.6 = { # Plebisciti
    type = country_event
    placement = ROOT

    title = italian_independence_war.6.t
    desc = italian_independence_war.6.d

    event_image = {
        video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
    }

    trigger = {
       #From je_italy_plebisciti JE
    }

    immediate = {
        set_global_variable = plebiscite_variable
        set_variable = plebiscite_var

        s:STATE_LAZIO = {
            add_claim = c:SAR
        }
        s:STATE_EMILIA = {
            add_claim = c:SAR
        }
        s:STATE_TUSCANY = {
            add_claim = c:SAR
        }
        s:STATE_ROMAGNA = {
            add_claim = c:SAR
        }
         s:STATE_UMBRIA = {
            add_claim = c:SAR
        }
    }

    option = {
        name = italian_independence_war.6.a
        default_option = yes
        ai_chance = {
            base = 100
        }

        s:STATE_LAZIO = {
            set_owner_of_provinces = {
                country = c:SAR
                provinces = { "x3040A0" "xB0C0A0" "xB5726E" }           ###   "xB040A0" not Rome
            }
        }
        s:STATE_EMILIA = {
            every_scope_state = {
                limit = {
                    owner = {
                        country_has_primary_culture = cu:north_italian
                        is_player = no
                    }
                }
                set_state_owner = c:SAR
            }
        }
        s:STATE_TUSCANY = {
            every_scope_state = {
                limit = {
                    owner = {
                        country_has_primary_culture = cu:north_italian
                        is_player = no
                    }
                }
                set_state_owner = c:SAR
            }
        }
        s:STATE_ROMAGNA = {
            every_scope_state = {
                limit = {
                    owner = {
                        country_has_primary_culture = cu:north_italian
                        is_player = no
                    }
                }
                set_state_owner = c:SAR
            }
        }
        s:STATE_UMBRIA = {
            every_scope_state = {
                limit = {
                    owner = {
                        country_has_primary_culture = cu:north_italian
                        is_player = no
                    }
                }
                set_state_owner = c:SAR
            }
        }

        # Cede Nice + Savoy
        if = {
            limit = {
                exists = s:STATE_SAVOY.region_state:SAR
                has_diplomatic_pact = {
                    who  = c:FRA
                    type = alliance
                }
            }
            s:STATE_SAVOY.region_state:SAR = { set_state_owner = c:FRA }
        }
        if = {
            limit = {
                exists = s:STATE_PROVENCE.region_state:SAR
                has_diplomatic_pact = {
                    who  = c:FRA
                    type = alliance
                }
            }
            s:STATE_PROVENCE.region_state:SAR = { set_state_owner = c:FRA }
        }
        change_infamy = -30

        ### Full alliance no longer needed ####
        if = {
            limit = {
                has_diplomatic_pact = {
                    who  = c:FRA
                    type = alliance
                }
            }
            c:SAR = {
                remove_diplomatic_pact = {
                    country = c:FRA
                    type = alliance
                }
            }
            c:SAR = {
                create_diplomatic_pact = {
                    country = c:FRA
                    type = defensive_pact
                }
            }
        }
    }

    option = {
        name =  italian_independence_war.6.b
        ai_chance = {
            base = 0
        }
    }

}
